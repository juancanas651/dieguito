from PIL import Image, ImageDraw, ImageFont

# Function to create a basic mockup image with text
def create_mockup(text, size=(800, 600), background_color=(200, 200, 200)):
    img = Image.new('RGB', size, color=background_color)
    draw = ImageDraw.Draw(img)
    
    # Use a basic font and draw text in the center
    try:
        font = ImageFont.truetype("arial.ttf", 40)
    except IOError:
        font = ImageFont.load_default()
    
    textwidth, textheight = draw.textsize(text, font)
    width, height = img.size
    x = (width - textwidth) / 2
    y = (height - textheight) / 2
    draw.text((x, y), text, font=font, fill=(0, 0, 0))
    
    return img

# Create mockup images for each section
